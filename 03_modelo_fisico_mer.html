Tabelas

CLIENTE

Campo / definição
id_cliente (PK) INT
nome VARCHAR(120)
email VARCHAR(120) UNIQUE
telefone VARCHAR(20)
cpf VARCHAR(14) UNIQUE
data_cadastro DATETIME
status VARCHAR(20)

ENDERECO

Campo / definição
id_endereco (PK) INT
id_cliente (FK) INT
logradouro VARCHAR(120)
numero VARCHAR(10)
complemento VARCHAR(80)
bairro VARCHAR(80)
cidade VARCHAR(80)
estado CHAR(2)
cep VARCHAR(9)
tipo_endereco VARCHAR(20)

CATEGORIA

Campo / definição
id_categoria (PK) INT
nome VARCHAR(80) UNIQUE
descricao VARCHAR(255)
status VARCHAR(20)

PRODUTO

Campo / definição
id_produto (PK) INT
id_categoria (FK) INT
nome VARCHAR(120)
descricao TEXT
sku VARCHAR(50) UNIQUE

estoque INT
ativo BOOLEAN
data_cadastro DATETIME

Campo / definição
id_fornecedor (PK) INT
razao_social VARCHAR(120)

telefone VARCHAR(20)
status VARCHAR(20)
